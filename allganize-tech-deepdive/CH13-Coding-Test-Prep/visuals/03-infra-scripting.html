# 1단계: 원시 로그 읽기 Read raw log
raw = "2024-03-15 14:32:01 ERROR [api] Connection timeout to db-master"
# 2단계: 정규식 추출 Regex extract
pattern = r"(\d{4}-\d{2}-\d{2} \d{2}:\d{2}:\d{2}) (\w+) \[(\w+)\] (.+)"
match = re.match(pattern, raw)
# 3단계: 구조화 Structured data
log_entry = {"ts": match[1], "level": match[2], "svc": match[3], "msg": match[4]}
# 4단계: 리포트 / 알림 Report or alert
if log_entry["level"] == "ERROR": send_alert(log_entry)
